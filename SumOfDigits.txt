namespace Assignment
{
    public class SumOfDigits
    {

        /// <summary>
        /// Gives you the sum of digits present in the number
        /// </summary>
        /// <param name="input"></param>
        public void getSumOfDigits(int input)
        { 
            int number = input;
            int sumOfDigits = 0;

            while (number > 0)
            {
                
                int reminder = number % 10;
                sumOfDigits = sumOfDigits + reminder;
                number = number / 10;
            }

            //printing the result
            Console.WriteLine("The sum of digits is : " + sumOfDigits);
        }
    }

}
