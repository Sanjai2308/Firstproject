using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;




namespace Assignment
{
    public class MathematicalTable
    {
        // Declaring the global variable to make it accesible and to initialize it according to the user input
        public int limit;

        /// <summary>
        /// Gives you the mathematical table for the passed inputs
        /// </summary>
        /// <param name="number"></param>
        
        // Creating the method to form an Table by calling it from the main class
        public void getMathematicalTable(int number )
            {
                // loop for achieving the output
                for (int firstNumber = 1; firstNumber <= limit; firstNumber++)
                {
                    int product = firstNumber * number;
                    Console.WriteLine(firstNumber + "x" + number + "=" + product);
                }

            }
       

    }
