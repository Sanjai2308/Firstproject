using System;
using System.Collections;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Assignment
{
    public class PrimeNumbers
    {
        /// <summary>
        /// Checks the prime numbers betwwen the range, stores in List and returns a list object
        /// </summary>
        /// <param name="limit"></param>
        /// <returns></returns>
        public List<int> getPrimeNumbers(int limit)
        {
            // Creating the List<int> to store prime numbers
            List<int> primeNumbers = new List<int>();   

            for (int number = 1; number <= limit; number++)
            {

                if (isPrime(number))
                {
                    primeNumbers.Add(number);

                }

            }
            // Returns a List object to the called method
            return primeNumbers;
        }


        /// <summary>
        /// Checks the passed number is Prime or not
        /// </summary>
        /// <param name="number"></param>
        /// <returns></returns>

        // This method checks whether the passed number is divisible by any of the number which is less than half of the passed number
        public  bool isPrime(int number) 
        {
            // Logic to check prime or not
            bool prime = true;
            for (int divider = 2; divider <= number/2; divider++)
            {
                if (number % divider == 0) 
                { 
                    prime = false;
                    break;
                }
            }

            // Returning the boolean value for the called method statement
            return prime;
        }
    }
}
